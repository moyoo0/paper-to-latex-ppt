% =========================================================
% 通用图片组件
% 只控制图片尺寸，不定义图片用途
% =========================================================

\newcommand{\fullwidthimage}[1]{%
    \begin{center}
        \includegraphics[
            width=0.95\textwidth,
            height=0.72\textheight,
            keepaspectratio
        ]{#1}
    \end{center}
}

\newcommand{\columnimage}[1]{%
    \begin{center}
        \includegraphics[
            width=\linewidth,
            height=0.66\textheight,
            keepaspectratio
        ]{#1}
    \end{center}
}

% =========================================================
% TikZ 通用组件样式
% 仅定义节点和箭头外观，不规定流程内容与数量
% =========================================================

\tikzset{
    pptbox/.style={
        rectangle,
        rounded corners=4pt,
        draw=DeepBlue,
        fill=LightGray,
        text=black!80,
        align=center,
        inner sep=8pt,
        font=\small
    },
    pptboxwide/.style={
        pptbox,
        text width=12cm
    },
    pptboxhighlight/.style={
        rectangle,
        rounded corners=4pt,
        draw=AccentRed,
        fill=LightGray,
        text=black!80,
        align=center,
        inner sep=8pt,
        font=\small
    },
    pptarrow/.style={
        ->,
        >=Stealth,
        thick,
        draw=DeepBlue
    },
    pptarrowaccent/.style={
        ->,
        >=Stealth,
        thick,
        draw=AccentRed
    }
}
